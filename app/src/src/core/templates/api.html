You need to: 
          import requests

          config = requests.get('https://account.hubstaff.com/.well-known/openid-configuration').json()
          token_endpoint = config['token_endpoint']

          # set up a refresh token manually on https://developer.hubstaff.com/personal_access_tokens
          refresh_token = 'your-refresh-token'
          access_data = requests.post(token_endpoint, data={'grant_type': 'refresh_token', 'refresh_token': refresh_token}).json()
          access_token = access_data['access_token']
          print('My access token:', access_token)
